# Run this code in a Python environment with Streamlit installed
# pip install streamlit pandas plotly openpyxl

import streamlit as st
import pandas as pd
import plotly.express as px

# 1. Page Configuration
st.set_page_config(page_title="CEM Dashboard", layout="wide")

# 2. Load Data (Replace with actual Excel path)
@st.cache_data
def load_data():
    # Logic to read Excel file
    # df = pd.read_excel("dataset.xlsx")
    
    # Generating Dummy Data for Demonstration
    data = {
        'CEM Number': [f'CEM-{i}' for i in range(1, 101)],
        'City Name': ['New York', 'London', 'Singapore', 'Dubai', 'Mumbai'] * 20,
        'Fesibility Name': [f'Feas-{i}' for i in range(100)],
        'Total H.P': [50 + i*5 for i in range(100)],
        'Active': [1, 0] * 50,
        'Not Active': [0, 1] * 50,
        'Allocated Budget': [10000 + i*1000 for i in range(100)],
        'Instant Budget Requirement': [5000 + i*500 for i in range(100)],
        '% of Completion of Work Done': [10 + (i%10)*8 for i in range(100)],
        'Days Remaining': [30 + (i%5)*10 for i in range(100)],
        'Status': ['Live' if i%2==0 else 'Not Live' for i in range(100)],
        'Timeline - Monthwise': ['Jan', 'Feb', 'Mar', 'Apr']*25
    }
    return pd.DataFrame(data)

df = load_data()

# 3. Sidebar Filters
st.sidebar.header("Dashboard Filters")

city = st.sidebar.multiselect("Select City", df['City Name'].unique(), default=df['City Name'].unique())
status = st.sidebar.selectbox("Select Status", ["All", "Live", "Not Live"])
month = st.sidebar.multiselect("Select Month", df['Timeline - Monthwise'].unique(), default=df['Timeline - Monthwise'].unique())

# Apply Filters
df_selection = df.query(
    "`City Name` == @city & `Timeline - Monthwise` == @month"
)
if status != "All":
    df_selection = df_selection[df_selection['Status'] == status]

# 4. KPI Section
total_projects = df_selection.shape[0]
total_budget = df_selection['Allocated Budget'].sum()
total_active = df_selection['Active'].sum()
avg_completion = df_selection['% of Completion of Work Done'].mean()

col1, col2, col3, col4 = st.columns(4)
with col1:
    st.metric("Total Projects", f"{total_projects:,}")
with col2:
    st.metric("Total Budget", f"${total_budget:,.0f}")
with col3:
    st.metric("Active Customers", f"{total_active:,}")
with col4:
    st.metric("Avg Completion", f"{avg_completion:.1f}%")

st.markdown("---")

# 5. Charts
col_left, col_right = st.columns(2)

with col_left:
    fig_budget = px.scatter(
        df_selection, x='Allocated Budget', y='% of Completion of Work Done',
        color='City Name', size='Total H.P', title="Budget vs Completion %"
    )
    st.plotly_chart(fig_budget, use_container_width=True)

with col_right:
    fig_city = px.bar(
        df_selection.groupby('City Name')['Active'].sum().reset_index(),
        x='City Name', y='Active', title="City-wise Active Customers",
        color_discrete_sequence=['#00A9F4']
    )
    st.plotly_chart(fig_city, use_container_width=True)

fig_timeline = px.line(
    df_selection, x='Timeline - Monthwise', y='% of Completion of Work Done',
    title="Timeline Progress (Monthwise)", markers=True
)
st.plotly_chart(fig_timeline, use_container_width=True)

# 6. Table
st.dataframe(df_selection)

# 7. Auto Refresh
count = st_autorefresh(interval=60000, limit=None, key="refresh")
